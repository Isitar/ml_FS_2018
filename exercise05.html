def standardize(npArr, mean = None, std = None):
    if (mean is None):
        mean = npArr.mean(axis=0)
    if (std is None):
        std = npArr.std(axis=0)
    npArr_std = (npArr - mean)/std
    return npArr_std, mean, std

Leiten sie den Gradienten der Maximum-Likelihood cost-function

$ L(\beta, X, y) = - \sum_{i=1}^{n} \Big( y^{(i)}\log(\phi(x^{(i)}\beta)) + (1 - y^{(i)})\log(1 - \phi(x^{(i)}\beta)) \Big) $

Student's answer Score: 1.0 / 2.0 (Top)
phi abgeleitet¶

$ \phi= {\dfrac{1}{(1+e^{-x})}} $

Zähler $ e^{-x} $

Nenner: $ (1+e^{-x})^2 = (e^{-x})^2+2*e^{-x}+1 $

$ \phi'= {\dfrac{e^{-x}}{(e^{-x})^2+2*e^{-x}+1}} $

1. Teil abgeleitet¶

$ {\dfrac{y}{\phi(\beta x)}}* \phi'(\beta x) $

2. Teil abgeleitet¶

$ {\dfrac{1-y}{1-\phi(\beta x)}} * -\phi'(\beta x) $

Zusammen¶

$ {\dfrac{y*\phi'(\beta x)}{\phi(\beta x)}} + {\dfrac{(1-y) * -\phi'(\beta x)}{1-\phi(\beta x)}} $

from scipy.optimize import minimize


class LogisticRegression(object):
    
      
    def __init__(self, alpha=0., method=None, tol=None):
        '''
        
        Parameters
        ----------
        
        :method: 'CG', 'L-BFGS-B', ..
        :alpha: Corresponds to the regularization parameter lambda
        '''
        self.alpha = alpha
        self.method = method
        self.tol = tol
        self.coef_ = np.zeros(0)

    def fit(self, X, y):        
        # init coefs
        if (self.coef_.shape != X.shape[1]):
            self.coef_ = np.zeros(X.shape[1])
            
        
        # todo: callback = collectBetas(X)
        res = minimize(self.cost_function, x0=self.coef_, args=(X, y, self.alpha),  method=self.method, tol=self.tol)
        self.coef_ = res.x
        return res
    
    @staticmethod
    def sigmoid(x):
        return 1/(1+np.exp(-x))
        
    @staticmethod
    def cost_function(beta, X, y, alpha):
        sum = 0
        for i in range(y.shape[0]):
            temp = y[i]*np.log(LogisticRegression.sigmoid(np.dot(X[i,],beta))) + (1-y[i])*np.log(1-LogisticRegression.sigmoid(np.dot(X[i,],beta)))
            sum -= temp
            
        sum += alpha * (beta**2).sum()
        return sum
        
    
    
    
    def predict(self, X):
        return X.dot(self.coef_) >= 0.5
    
    def score(self, X, y, method='accuracy', verbose=False):       
        yPredicted = self.predict(X)
        yBool = y.astype(bool)
        yPredictedBool = yPredicted.astype(bool)
        TP = np.logical_and(yBool, yPredictedBool).sum()
        FP = np.logical_and(yBool == False, yPredictedBool).sum()
        
        TN = np.logical_and(yBool == False, yPredictedBool == False).sum()
        FN = np.logical_and(yBool, yPredictedBool == False).sum()
        
        PredictedT = yPredictedBool.sum()
        PredictedN = (yPredictedBool == False).sum()
        
        SampleT = yBool.sum()
        SampleN = (yBool == False).sum()
        
        Total = y.shape[0]
        if (verbose):
            print('TP: {TP}, FP: {FP}, TN: {TN}, FN: {FN}'.format(TP=TP, FP=FP, TN=TN, FN=FN))
        
        if (method == 'accuracy'):
            return (TP + TN)/Total
        if (method == 'precision'):
            return TP / PredictedT
        if (method == 'recall'):
            return TP / SampleT

from matplotlib.colors import ListedColormap

def plot_decision_regions(X, y, classifier, resolution=0.02, x_label = '', y_label='', title = ''):
    markers = ('o', 'o', 'o', 'o', 'o')
    colors = ('red', 'blue', 'lightgreen', 'gray', 'cyan')
    cmap = ListedColormap(colors[:len(np.unique(y))])
    
    # plot the decision surface
    x1_min, x1_max = X[:, 0].min() - 1, X[:, 0].max() + 1
    x2_min, x2_max = X[:, 1].min() - 1, X[:, 1].max() + 1
    xx1, xx2 = np.meshgrid(np.arange(x1_min, x1_max, resolution),
    np.arange(x2_min, x2_max, resolution))
    Z = classifier.predict(np.array([xx1.ravel(), xx2.ravel()]).T)
    Z = Z.reshape(xx1.shape)
    fig, ax = plt.subplots(figsize=(6,6))
    contour = ax.contourf(xx1, xx2, Z, alpha=0.4, cmap=cmap)
    ax.set_xlim(xx1.min(), xx1.max())
    ax.set_ylim(xx2.min(), xx2.max())
    
    # plot all samples
    for idx, cl in enumerate(np.unique(y)):
        ax.scatter(x=X[y == cl, 0], y=X[y == cl, 1],
        alpha=0.8, c=cmap(idx),
        marker=markers[idx], label=cl,
        edgecolor='none')
    _ = ax.set_xlabel(x_label)
    _ = ax.set_ylabel(y_label)
    _ = ax.set_title(title)

class Perceptron:
    '''
    Das Perceptron ist eines der ersten einfachen Modelle welches die Funktionsweise eines Neurons in einem
Algorithmus implementiert:

    
    '''
    
    
    def __init__(self, max_steps):        
        self.max_steps = max_steps
        self.omegas = np.zeros(3)
    @staticmethod
    def encodeOutput(e, matcher):
        if (e == matcher):
            return 1
        return -1
    
    def fit(self, X, y, classToFind, verbose=False):
        numberOfData = y.shape[0]
        
        preparedY = y.copy()
        if (verbose):
            print('y before encoding: {y}'.format(y=preparedY))
        for i in range(numberOfData):
            preparedY[i] = self.encodeOutput(e=preparedY[i], matcher=classToFind)
        
        if (verbose):
            print('y after encoding: {y}'.format(y=preparedY))

        # add bias
        preparedX = self.add_bias_input(X)
        if (verbose):
            print('prepared X: {X} \n shape: {shape}'.format(X=preparedX, shape=preparedX.shape))
        
        self.omegas = np.zeros(X.shape[1] + 1) #bias
        if (verbose):
            print('init omegas: {omega}'.format(omega=self.omegas))
          
        n = 0.5
        self.step = 0
        self.has_error = True
        while (not self.do_stop()):
            self.has_error = False
            for i in range(preparedY.shape[0]):
                y_hat = self.predict(preparedX[i])
                self.has_error = self.has_error or (y_hat != preparedY[i])
                if (verbose):
                    print('y_hat: {y_hat}, real y: {y}'.format(y_hat=y_hat, y=preparedY[i]))
                self.omegas += n*(preparedY[i] - y_hat)*preparedX[i]
            self.step += 1
        print('terminated after {steps} steps. has_errors: {has_err}'.format(steps=self.step, has_err = self.has_error))
        
    def do_stop(self):
        if (self.step >= self.max_steps):
            return True;
        if (not self.has_error): 
            return True
        return False;
    
    
    def predict(self, X):
        predictorX = X.copy()
        try:
            if (predictorX.shape[1] != self.omegas.shape[0]):
                predictorX = self.add_bias_input(predictorX);
        except IndexError:
            None #ignore
            
        val = predictorX.dot(self.omegas.T)
        if (isinstance(val, float)):
            return self.activation_function(val)
        
        retVal = val.copy()
        for i in range(retVal.shape[0]):
            retVal[i] = self.activation_function(retVal[i])
        
        return retVal
        
    
    @staticmethod
    def activation_function(x):
        if (x > 0.):
            return 1
        return -1
    
    @staticmethod
    def add_bias_input(X):
        return np.hstack((np.ones((X.shape[0],1)), X))
