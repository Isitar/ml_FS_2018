def generate_random_linear_data(n_points=1000, b0=6.0, b1=1.5, sigma=0.1, mux=0.5):
    '''
    Generates uniformiformly distributed x with mean mux and
    y according to a linear model.
    '''
    x = np.random.rand(n_points)+mux-0.5
    y = b0 + x*b1 + sigma*np.random.randn(n_points)
    return x, y

class LinearRegressionGradientDescent(object):
    '''
    Implements a Gradient Descent Optimization for a Simple 
    Linear Regression Problem. 
    '''
    
    def __init__(self, init_coef=(0., 0.), epsilon=0.00001, maxsteps=1000, 
                 linesearch='fixed', stepsize=0.001, 
                 alpha = 0.1, beta = 0.7):
        '''
        Gradient Descent Optimizer for Linear Regression.
        '''
        self.init_coef = init_coef
        self.coef_ = np.array(self.init_coef)
        self._nsteps = 0
        self.stepsize = stepsize
        self.epsilon = epsilon
        self.maxsteps = maxsteps
        self.linesearch = linesearch

        # YOUR CODE HERE
        
        self.normgrad_ = []
        self.cost_function_values = [];
        #print('LinearRegressionGradientDescent(init_coef={}, epsilon={},\nmaxsteps={},'.format(self.init_coef, self.epsilon, self.maxsteps) +' linesearch={}, stepsize={})\n'.format(self.linesearch,self.stepsize))
    
    
    def fit(self, X, y, verbose=False, btAlpha = 0.01, btBeta = 0.38):
        '''
        Fits the coefficients beta of a linear regression problem
        to the dataset (X, y).
        '''
        
        
        # YOUR CODE HERE
        self.normgrad_ = []
        self.cost_function_values = [];
        self._nsteps = 0
        
        while not self._do_stop(X, y):
            
            neg_grad = -self.least_squares_gradient(self.coef_, X, y)            
            if self.linesearch == 'fixed':
                self.coef_ += self.stepsize*neg_grad
                
            elif self.linesearch == 'backtracking':
                stepsize = self.backtracking_linesearch(X, y, alpha = btAlpha, beta = btBeta)
                self.coef_ += stepsize*neg_grad
                
            # YOUR CODE HERE
            self.cost_function_values.append(self.cost_function(X,y,self.coef_))
                        
            if verbose:
                print('step {s} : neg grad = {ng}, normgrad: {nog},  coef_ = {c}, stepsize = {ss}'.format(
                        s=self._nsteps, ng=neg_grad, c=self.coef_,
                        nog=np.linalg.norm(neg_grad),
                        ss=self.stepsize if self.linesearch == 'fixed' else stepsize ))

            self._nsteps += 1
        
        if (verbose):
            
            print('Done fitting!')
                
            print('n steps : ', self._nsteps)
            print('norm gradient : ', self.normgrad_[-1])
        
        return self
                    
    def _do_stop(self, X, y):
        neg_grad = -self.least_squares_gradient(self.coef_, X, y)
        norm_neg_grad = np.linalg.norm(neg_grad)
        self.normgrad_.append(norm_neg_grad)
        tiny = norm_neg_grad < self.epsilon        
        manysteps = self._nsteps >= self.maxsteps
        return (tiny or manysteps)
    
    
    def predict(self, X):
        return X.dot(self.coef_)
    
    
    @staticmethod
    def least_squares_gradient(coef, X, y):
        '''
        Calculates the least squares gradient at position coef
        for the dataset (X, y).
        '''
        grad = []
        for idx in range(X.shape[1]):
            grad.append(((y - coef.dot(X.T))*X[:, idx]).sum())
        return -2*np.array(grad)
    
    
    def score(self, X, y):
        return 1- ((X - X.mean())**2).sum()/((y-y.mean())**2).sum()
        
        
    @staticmethod
    def cost_function(X, y, beta):
        return ((y - (beta[0] + beta[1]*X[:,1]))**2).sum()
        #old code, more readable but slow
        # J(b0, b1) = sum((yi - b0+b1x1i)^2)
        #retVal = 0;
        #for i in range(y.size):
        #    retVal += (y[i] - (beta[0] + beta[1]*X[i,1]))**2 
        #return retVal
            
    def backtracking_linesearch(self, X, y, alpha = 0.01, beta = 0.38):
        #n = 1
        #solange f(x-n*grad f(x)) > f(x) - alpha * n * grad fx)^T * Grad f(x)
        #  n = beta * n
        
        coef = self.coef_
            
        n = 1;
        
        fx = self.cost_function(X,y,coef)
        gradFx = self.least_squares_gradient(coef,X,y);
        
        while (self.cost_function(X,y,coef - n * self.least_squares_gradient(coef,X,y)) > (fx  - alpha * n * np.dot(gradFx.transpose(), gradFx))):
            n = beta * n;
        return n;

def create_contour_plot(X, y, lims=(-10., 10.), resolution=.2, ax=None):
    # plot parameters
    b0_min, b0_max = lims
    b1_min, b1_max = lims

    # create meshgrid
    xx1, xx2 = np.meshgrid(np.arange(b0_min, b0_max, resolution),
                         np.arange(b1_min, b1_max, resolution))

    Z = []
    for b0, b1 in zip(xx1.ravel(), xx2.ravel()):
        Z.append(LinearRegressionGradientDescent.cost_function(X, y, [b0,b1]))
